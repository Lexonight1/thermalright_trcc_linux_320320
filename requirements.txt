# TRCC Linux GUI - Python Dependencies
# Install: pip install -r requirements.txt

# ==============================================================================
# Python Packages
# ==============================================================================

# Core (required)
Pillow>=10.0.0          # Image processing (PNG, JPEG, BMP)
psutil>=5.9.0           # System monitoring (CPU, RAM, temps)
requests>=2.28.0        # Cloud theme downloads

# System Tray (optional but recommended)
pystray>=0.19.0         # System tray icon (matches Windows TRCC NotifyIcon)

# Video playback (optional - alternative to ffmpeg system package)
# NOTE: ffmpeg system package is RECOMMENDED (matches Windows TRCC behavior)
# opencv-python is an alternative if ffmpeg is not available
opencv-python>=4.8.0    # Video frame extraction (alternative to ffmpeg)

# ==============================================================================
# System Packages (install via package manager)
# ==============================================================================
#
# REQUIRED:
#   sg3-utils     - Provides sg_raw for SCSI/USB LCD communication
#   lsscsi        - Maps USB devices to /dev/sgX SCSI devices
#   usbutils      - Provides lsusb for USB device detection
#   python3-tk    - Tkinter GUI toolkit
#
# RECOMMENDED (for video playback - matches Windows TRCC):
#   ffmpeg        - Video frame extraction (Windows TRCC uses ffmpeg directly)
#
# ------------------------------------------------------------------------------
# Installation by Distribution:
# ------------------------------------------------------------------------------
#
# Fedora/RHEL/CentOS:
#   sudo dnf install sg3_utils lsscsi usbutils python3-tkinter
#   sudo dnf install ffmpeg ffmpeg-free   # optional, for video
#
# Ubuntu/Debian:
#   sudo apt install sg3-utils lsscsi usbutils python3-tk
#   sudo apt install ffmpeg               # optional, for video
#
# Arch Linux:
#   sudo pacman -S sg3_utils lsscsi usbutils tk
#   sudo pacman -S ffmpeg                 # optional, for video
#
# openSUSE:
#   sudo zypper install sg3_utils lsscsi usbutils python3-tk
#   sudo zypper install ffmpeg            # optional, for video
#
# ==============================================================================
